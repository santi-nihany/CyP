Monitor BaseDeDatos {
    int libre = 5, esperando = 0;
    cond espera;
    procedure entrar() {
        if(libre == 0){
            esperando++;
            wait(espera);
        }
        libre--;
    }

    procedure salir(){
        if(esperando > 0 ){
            esperando--;
            signal(espera);
        } else {
            libre++;
        }
    }

}
